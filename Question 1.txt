CREATE TABLE `product` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`sku_id` int(8) NOT NULL,x
`name` varchar(255) NOT NULL,
`price` int(10) NOT NULL,
`cost` int(10) NOT NULL,
`on_sale` tinyint(1) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`),
KEY `FK` (`sku_id`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `product_gallery` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`product_id` int(8) NOT NULL,x
`asset_url` varchar(255) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`),
KEY `FK` (`product_id`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `inventory` (
`sku` int(11) unsigned NOT NULL AUTO INCREMENT,
`name` varchar(255) NOT NULL,
`location_id` int(10) NOT NULL,
`category_id` int(10) NOT NULL,
`stock_count` int(10) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`sku`),
KEY `FK` (`location_id`),
KEY `FK` (`category_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `location` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`name` varchar(255) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `category` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`name` varchar(255) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `orders` (
`order_id` int(11) unsigned NOT NULL AUTO INCREMENT,
`total_price` int(10) NOT NULL,
`placed_orders` json NOT NULL,
`status` varchar(20) NOT NULL,
`payment_date` timestamp NULL DEFAULT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`order_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `users` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`name` varchar(100) NOT NULL,
`email` varchar(100) NOT NULL,
`password` varchar(255) NOT NULL,
`payment_date` timestamp NULL DEFAULT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `user_role` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`user_id` int(10) NOT NULL,
`role_id` int(10) NOT NULL,
PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `role` (
`id` int(11) unsigned NOT NULL AUTO INCREMENT,
`name` varchar(100) NOT NULL,
`display_name` varchar(100) NOT NULL,
`description` varchar(100) NOT NULL,
`created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
`updated` timestamp NULL DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP,
PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
